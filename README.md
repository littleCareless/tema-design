## Install
```shell
npm install tema-ui -S
```

## Quick Start
``` javascript
import Vue from 'vue'
import TemaUi from 'tema-ui'

Vue.use(TemaUi)

// or
import {
  Select,
  Button
  // ...
} from 'tema-ui'

Vue.component(Select.name, Select)
Vue.component(Button.name, Button)
```
